
[size=20][b]Обзор[/b][/size]

Консольные переменные (ConVars) являются основным способом настройки параметров игры и поведения плагинов. С новой поддержкой классов вы можете работать с ConVars двумя способами:

[list]
[*][b]Подход на основе классов[/b] - Использование класса-обёртки [font=monospace]ConVar[/font] для более чистого объектно-ориентированного кода
[*][b]Подход на основе функций[/b] - Традиционные функции в стиле C с явными дескрипторами
[/list]

[quote][b]Info:[/b] [b]Новое[/b]: Класс [font=monospace]ConVar[/font] обеспечивает автоматическое управление ресурсами, более чистый синтаксис и лучшую поддержку IDE. Используйте его для нового кода, если он доступен на вашем языке![/quote]

[size=20][b]Поиск ConVar[/b][/size]

[spoiler=С классами (рекомендуется)]
Используйте статический метод [font=monospace]ConVar.Find[/font] для получения экземпляра ConVar:
  [code=csharp]
  var cheatsCvar = ConVar.Find("sv_cheats");
  if (cheatsCvar != null && cheatsCvar.Valid()) {
      // Используйте экземпляр ConVar
  }
  
[/code]
[/spoiler]

[spoiler=Без классов]
Используйте функцию [font=monospace]FindConVar[/font] для получения дескриптора:
  [code=csharp]
  var cheatsCvar = FindConVar("sv_cheats");
  
[/code]
[/spoiler]

[size=20][b]Манипулирование примитивными значениями[/b][/size]

[spoiler=С классами (рекомендуется)]
Чтение и установка значений более интуитивны с методами:
  [code=csharp]
  var cheatsCvar = ConVar.Find("sv_cheats");

  // Чтение значения с использованием типизированных методов
  bool value = cheatsCvar.GetBool();

  // Установка значения с использованием типизированных методов
  cheatsCvar.SetBool(true);

  // Вы также можете использовать универсальный метод Set
  cheatsCvar.Set(true);
  
[/code]
[/spoiler]

[spoiler=Без классов]
Чтение и установка требует явной передачи дескриптора:
  [code=csharp]
  var cheatsCvar = FindConVar("sv_cheats");

  // Чтение значения
  bool value = GetConVarBool(cheatsCvar);

  // Установка значения с дополнительными параметрами
  SetConVarBool(cheatsCvar, true, /*replicate*/true, /*notify*/false);

  // Или используйте упрощённый помощник
  SetConVar(cheatsCvar, true);
  
[/code]
[/spoiler]

[size=20][b]Манипулирование объектами[/b][/size]

[spoiler=С классами (рекомендуется)]
Строковые значения обрабатываются через типизированные методы:
  [code=csharp]
  var stringCvar = ConVar.Find("sv_skyname");

  // Получение строкового значения
  string skyName = stringCvar.GetString();
  PrintToServer($"sv_skyname = {skyName}");

  // Установка строкового значения
  stringCvar.SetString("foobar");
  
[/code]
[/spoiler]

[spoiler=Без классов]
Строковые значения требуют явного преобразования типов:
  [code=csharp]
  var stringCvar = FindConVar("sv_skyname");

  // Получение строкового значения
  string skyName = GetConVarString(stringCvar);
  PrintToServer($"sv_skyname = {skyName}");

  // Установка строкового значения
  SetConVarString(stringCvar, "foobar");
  
[/code]
[/spoiler]

[size=20][b]Создание ConVars[/b][/size]

[spoiler=С классами (рекомендуется)]
Создавайте ConVars используя конструктор класса с автоматическим определением типа:
  [code=csharp]
  // Создание различных типов ConVars
  var stringCvar = new ConVar("my_plugin_name", "DefaultName", "Player name", ConVarFlags.None);
  var intCvar = new ConVar("my_plugin_maxplayers", 32, "Max players", ConVarFlags.None);
  var floatCvar = new ConVar("my_plugin_speed", 1.5f, "Speed multiplier", ConVarFlags.None);
  var boolCvar = new ConVar("my_plugin_enabled", true, "Enable plugin", ConVarFlags.None);

  // Использование созданных ConVars
  if (boolCvar.GetBool()) {
      int maxPlayers = intCvar.GetInt32();
      PrintToServer($"Plugin enabled with {maxPlayers} max players\n");
  }
  
[/code]
[/spoiler]

[spoiler=Без классов]
Создавайте ConVars используя функции для конкретных типов:
  [code=csharp]
  // Создание различных типов ConVars
  var stringCvar = CreateConVarString("my_plugin_name", "DefaultName", "Player name", ConVarFlags.None);
  var intCvar = CreateConVarInt32("my_plugin_maxplayers", 32, "Max players", ConVarFlags.None);
  var floatCvar = CreateConVarFloat("my_plugin_speed", 1.5f, "Speed multiplier", ConVarFlags.None);
  var boolCvar = CreateConVarBool("my_plugin_enabled", true, "Enable plugin", ConVarFlags.None);

  // Использование созданных ConVars
  if (GetConVarBool(boolCvar)) {
      int maxPlayers = GetConVarInt32(intCvar);
      PrintToServer($"Plugin enabled with {maxPlayers} max players\n");
  }
  
[/code]
[/spoiler]

[size=20][b]Полный пример[/b][/size]

[spoiler=С классами (рекомендуется)]
[code=csharp]
  using Plugify;
  using static s2sdk.s2sdk;

  public unsafe class ConVarDemo : Plugin
  {
      private ConVar speedCvar;
      private ConVar gravityCvar;

      public void OnPluginStart()
      {
          // Создание пользовательских ConVars
          speedCvar = new ConVar("demo_speed", 1.0f, "Player speed multiplier", ConVarFlags.None);
          gravityCvar = new ConVar("demo_gravity", 1.0f, "Gravity multiplier", ConVarFlags.None);

          // Подключение к изменениям ConVar
          speedCvar.HookChange(OnSpeedChanged);

          // Поиск существующих игровых ConVars
          var cheatsCvar = ConVar.Find("sv_cheats");
          if (cheatsCvar != null && cheatsCvar.Valid())
          {
              bool cheatsEnabled = cheatsCvar.GetBool();
              PrintToServer($"Cheats enabled: {cheatsEnabled}\n");
          }
      }

      public void OnPluginEnd()
      {
          // Отключение от изменений
          if (speedCvar != null && speedCvar.Valid())
          {
              speedCvar.UnhookChange(OnSpeedChanged);
          }
      }

      private static void OnSpeedChanged(ulong handle, string oldValue, double newValue)
      {
          PrintToServer($"Speed changed from {oldValue} to {newValue}\n");
      }
  }
  
[/code]
[/spoiler]

[spoiler=Без классов]
[code=csharp]
  using Plugify;
  using static s2sdk.s2sdk;

  public unsafe class ConVarDemo : Plugin
  {
      private ulong speedCvar;
      private ulong gravityCvar;

      public void OnPluginStart()
      {
          // Создание пользовательских ConVars
          speedCvar = CreateConVarFloat("demo_speed", 1.0f, "Player speed multiplier", ConVarFlags.None);
          gravityCvar = CreateConVarFloat("demo_gravity", 1.0f, "Gravity multiplier", ConVarFlags.None);

          // Подключение к изменениям ConVar
          HookConVarChange(speedCvar, OnSpeedChanged);

          // Поиск существующих игровых ConVars
          ulong cheatsCvar = FindConVar("sv_cheats");
          if (cheatsCvar != 0)
          {
              bool cheatsEnabled = GetConVarBool(cheatsCvar);
              PrintToServer($"Cheats enabled: {cheatsEnabled}\n");
          }
      }

      public void OnPluginEnd()
      {
          // Отключение от изменений
          if (speedCvar != 0)
          {
              UnhookConVarChange(speedCvar, OnSpeedChanged);
          }
      }

      private static void OnSpeedChanged(ulong handle, string oldValue, double newValue)
      {
          PrintToServer($"Speed changed from {oldValue} to {newValue}\n");
      }
  }
  
[/code]
[/spoiler]
